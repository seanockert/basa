<link rel="stylesheet" href="css/responsive-nav.css" type="text/css">
<script src="js/plugins/responsive-nav.min.js"></script>

<!-- This hamburger button will show on mobile instead of the full menu: -->
<button class="open nav-menu x" type="button" role="button" id="menu" aria-label="Toggle Navigation">
  <span class="lines"></span>
</button>

<!-- Wrap the navigation in a list like so: -->
<nav id="responsive-nav" class="nav-collapse">
  <ul>
    <li><a href="#">Home</a></li>
    <li><a href="#">Services</a></li>
    <li><a href="#">About</a></li>
    <li><a href="#">Contact</a></li>
  </ul>
</nav>

<script>
  // Basic usage - provides a Menu link
  //var navigation = responsiveNav('#responsive-nav');

  // Advanced usage with custom button
  var navigation = responsiveNav('#responsive-nav', {
    transition: 250,
    label: 'Menu',
    insert: 'before',
    customToggle: '#menu',
    openPos: 'relative',
    open: function () {
      var menu = document.getElementById("menu");
      menu.className=menu.className.replace(/(^|\s)open(\s|$)/, 'close ');
    },
    close: function () {
      var menu = document.getElementById("menu");
      menu.className=menu.className.replace(/(^|\s)close(\s|$)/, 'open ');
    }
  });
</script>
