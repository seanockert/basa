How to use

  1. Attach the script js/plugins/responsive-nav.min.js
  2. Use the base styling in scss/plugins/_responsive-nav.scss
  3. Init the plugin with a nav ID reference: var navigation = responsiveNav('#responsive-nav');
  4. The breakpoint is defined in the CSS file. Default 40em/640px

Advanced usage with a custom button

var navigation = responsiveNav('#responsive-nav', {
  transition: 250,
  label: 'Menu',
  insert: 'before',
  customToggle: '#menu',
  openPos: 'relative',
  open: function () {
    var menu = document.getElementById("menu");
    menu.className=menu.className.replace(/(^|\s)open(\s|$)/, 'close ');
  },
  close: function () {
    var menu = document.getElementById("menu");
    menu.className=menu.className.replace(/(^|\s)close(\s|$)/, 'open ');
  }
});
