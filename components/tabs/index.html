How to use

SCSS: tabs.scss
Javascript: tabs.js 

<script src="js/plugins/tabs.js"></script>
<style>
/* Tabs required styles - https://24ways.org/2015/how-tabs-should-work/ */

ul.tabs {
  margin: 0;
  padding: 0;
  list-style: none;
  overflow: hidden;
  display: none;
}

  ul.tabs a,
  .tabs-panel a.tab {
    float: left;
    display: block;
    text-decoration: none;
  }

  ul.tabs a.selected,
  ul.tabs a[aria-selected='true'],
  .tabs-panel a.tab.selected,
  .tabs-panel a.tab[aria-selected='true'] {
    background-color: #ccc !important;
  }

.tabs-panel a.tab {
  width: 100%;
}

  .tabs-panel a.tab:after {
    display: block;
    content: '+';
    float: right;
  }

  .tabs-panel a.tab[aria-selected='true']:after,
  .tabs-panel a.tab.selected:after {
    content: '-';
  }

div[aria-hidden='true'] { display: none; }

@media only screen and (min-width: 40em) {
  ul.tabs { display: block; }

  .tabs-panel a.tab { display: none; }
}

/* End required styles */

  ul.tabs a,
  .tabs-panel a.tab {
    border: 1px solid #ccc;
    padding: 12px;
    margin-right: 6px;
    color: #000;
    line-height: 1;
  }

    ul.tabs a { border-bottom: none; }
    .tabs-panel a.tab { margin-bottom: 12px; }

    ul.tabs a:hover,
    .tabs-panel a.tab:hover { background-color: #eee; }

@media only screen and (min-width: 40em) {
  .tabs-panel {
    border-top: 1px solid #ccc;
    padding: 12px;
  }
}
</style>

<!-- Add the corresponding anchor links to the tabs to link them to the sections -->
<ul role="tablist" class="tabs">
  <li><a href="#first" class="tab" role="tab" aria-controls="first">First Tab</a></li>
  <li><a href="#second" class="tab" role="tab" aria-controls="second">Second Tab</a></li>
  <li><a href="#third" class="tab" role="tab" aria-controls="third">Third Tab</a></li>
</ul>

<!-- Tabbed sections -->
<div class="tabs-panel">
  <a href="#first" class="tab" aria-controls="first">First tab</a> <!-- Accordion headers only visible for mobile -->
  <div role="tabpanel" id="first">
    <h3>First tab content here</h3>
  </div>
  <a href="#second" class="tab" aria-controls="second">Second tab</a>
  <div role="tabpanel" id="second">
    <h3>Second tab content here</h3>
  </div>
  <a href="#third" class="tab" aria-controls="third">Third tab</a>
  <div role="tabpanel" id="third">
    <h3>Third tab content here</h3>
  </div>
</div>
