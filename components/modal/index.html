How to use

CSS: components-modal.css
Javascript: js/plugins/modal.js
JS dependency: js/vendors/classie.js 

<link rel="stylesheet" href="components-modal.css" type="text/css">
<script src="js/plugins/basa-modal.js"></script>

<!-- Add a link to open the modal with the anchor as the modal ID -->
<a href="#modal-1" class="modal-link">Open Modal</a>

<!-- Modal is hidden until link is clicked -->
<div id="modal-1" class="modal" aria-hidden="true">
  <p>Content</p>
  <a href="#" class="modal-close close">✕</a>
</div>

<div class="modal-overlay" class="modal-close></div>
